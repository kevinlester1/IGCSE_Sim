Array Visualization

Pseudo-code

procedure bubbleSort(array A)
n = length of A
repeat
swapped = false
for i from 1 to n-1
if A[i-1] > A[i] then
swap(A[i-1], A[i])
swapped = true
end if
end for
n = n - 1
until not swapped
end procedure
